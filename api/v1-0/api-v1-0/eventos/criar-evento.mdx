---
api: "POST /api/v1.0/events"
title: "Criar Evento"
---

Cria um novo evento.

<Request method="POST" url="/api/v1.0/events" auth="bearer" />

## Body params

<ParamField body="name" type="string" required>
  Nome do evento
</ParamField>

<ParamField body="location_type" type="string" required>
  Tipo de localização do evento. Valores aceitos:
  - "online": Evento online (city e state serão preenchidos automaticamente com "ONLINE")
  - "physical": Evento presencial (city e state são obrigatórios)
</ParamField>

<ParamField body="city" type="string">
  Cidade do evento (obrigatório se location_type for "physical")
</ParamField>

<ParamField body="state" type="string">
  Estado do evento - UF brasileira (obrigatório se location_type for "physical"). Valores aceitos: AC, AL, AP, AM, BA, CE, DF, ES, GO, MA, MT, MS, MG, PA, PB, PR, PE, PI, RJ, RN, RS, RO, RR, SC, SP, SE, TO
</ParamField>

<ParamField body="category" type="string">
  Categoria do evento. Valores aceitos: Esporte, Educação, Corporativo, Sociocultural
</ParamField>

<ParamField body="description" type="string">
  Descrição do evento
</ParamField>

<ParamField body="initial_date" type="string">
  Data inicial do evento (formato ISO 8601)
</ParamField>

<ParamField body="finish_date" type="string">
  Data final do evento (formato ISO 8601)
</ParamField>

<ParamField body="official_url" type="string">
  URL oficial do evento
</ParamField>

<ParamField body="image_url" type="string">
  URL da imagem do evento
</ParamField>

<ParamField body="image_key" type="string">
  Chave da imagem do evento (S3)
</ParamField>

<ParamField body="is_private" type="boolean">
  Se o evento é privado
</ParamField>

<ParamField body="date_mode" type="string">
  Modo de data do evento (single_date, date_range, recurring)
</ParamField>

<ParamField body="start_time" type="string">
  Horário de início do evento
</ParamField>

<ParamField body="end_time" type="string">
  Horário de término do evento
</ParamField>

<ParamField body="recurrence_type" type="string">
  Tipo de recorrência do evento. Valores aceitos: Semanal, Quinzenal, Mensal, Bimestral, Trimestral, Semestral, Anual. O valor será salvo em inglês no banco de dados.
</ParamField>

<ParamField body="recurrence_start_date" type="string">
  Data de início da recorrência
</ParamField>

<ParamField body="connected_event_ids" type="array">
  IDs de eventos conectados
</ParamField>

## Response

<ResponseField name="event_id" type="string">
  ID do evento criado
</ResponseField>

<ResponseField name="name" type="string">
  Nome do evento
</ResponseField>

<ResponseField name="description" type="string">
  Descrição do evento
</ResponseField>

<ResponseField name="category" type="string">
  Categoria do evento
</ResponseField>

<ResponseField name="subcategory" type="string">
  Subcategoria do evento
</ResponseField>

<ResponseField name="city" type="string">
  Cidade do evento
</ResponseField>

<ResponseField name="state" type="string">
  Estado (UF)
</ResponseField>

<ResponseField name="initial_date" type="string">
  Data/hora de início (ISO 8601)
</ResponseField>

<ResponseField name="finish_date" type="string">
  Data/hora de término (ISO 8601)
</ResponseField>

<ResponseField name="image_url" type="string">
  URL da imagem do evento
</ResponseField>

<ResponseField name="official_url" type="string">
  URL oficial do evento
</ResponseField>

<ResponseField name="status" type="string">
  Status do evento (ex: PENDING, ACTIVE)
</ResponseField>

<ResponseField name="created_at" type="string">
  Data/hora de criação (ISO 8601)
</ResponseField>

<ResponseField name="updated_at" type="string">
  Data/hora de atualização (ISO 8601)
</ResponseField>
