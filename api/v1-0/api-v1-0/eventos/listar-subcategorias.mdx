---
api: "GET /api/v1.0/{userCode}/subcategories"
title: "Listar Subcategorias por Template e Categoria"
---

Lista todas as subcategorias de uma categoria específica dentro de um template, usando os nomes do template e da categoria.

<Request method="GET" url="/api/v1.0/{userCode}/subcategories" auth="bearer" />

<ParamField path="userCode" type="string" required placeholder="ABC123XYZ">
  Código do usuário usado no path da rota.
</ParamField>

<ParamField query="template" type="string" required>
  Nome do template. Valores aceitos:
  - Cursos e Treinamentos
  - Evento Interno
  - Reconhecimento e Destaques
  - Evento Personalizado
  - Comunidades e Participações
  - Competições e Premiações
  - Certificações Profissionais
</ParamField>

<ParamField query="category" type="string" required>
  Nome da categoria. Valores aceitos:
  - Esporte
  - Educação
  - Corporativo
  - Sociocultural
</ParamField>

<Expandable title="properties">
  <ResponseField name="template" type="object">
    Informações do template
  </ResponseField>
  
  <ResponseField name="category" type="object">
    Informações da categoria
  </ResponseField>
  
  <ResponseField name="data" type="array">
    Lista de subcategorias da categoria
  </ResponseField>
  
  <ResponseField name="total" type="number">
    Total de subcategorias encontradas
  </ResponseField>
</Expandable>
