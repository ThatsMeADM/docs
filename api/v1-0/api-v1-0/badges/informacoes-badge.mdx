---
api: "GET /api/v1.0/badges/{id}"
title: "Informações de uma Badge"
---

Retorna um badge específico.

<Request method="GET" url="/api/v1.0/badges/{id}" auth="bearer" />

## Path params

<ParamField path="id" type="string" required>
  ID do badge
</ParamField>

## Response

<ResponseField name="badge_id" type="string">
  ID do badge
</ResponseField>

<ResponseField name="name" type="string">
  Nome do badge
</ResponseField>

<ResponseField name="description" type="string">
  Descrição do badge
</ResponseField>

<ResponseField name="created_at" type="string">
  Data/hora de criação (ISO 8601)
</ResponseField>

<ResponseField name="assets" type="object">
  Assets do badge
  <Expandable title="properties">
    <ResponseField name="image_url" type="string">
      URL da imagem do badge
    </ResponseField>
  </Expandable>
</ResponseField>

<ResponseField name="user" type="object">
  Informações do criador do badge
  <Expandable title="properties">
    <ResponseField name="profile" type="object">
      Perfil do criador
      <Expandable title="properties">
        <ResponseField name="profile_id" type="string">
          ID do perfil
        </ResponseField>

        <ResponseField name="name" type="string">
          Nome
        </ResponseField>

        <ResponseField name="company_name" type="string">
          Nome da empresa
        </ResponseField>
      </Expandable>
    </ResponseField>
  </Expandable>
</ResponseField>
