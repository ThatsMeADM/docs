---
api: "POST /api/v1.0/assertions"
title: "Emitir Certificados"
---

Emite certificados.

<Request method="POST" url="/api/v1.0/assertions" auth="bearer" />

## Body params

<ParamField body="badge_id" type="string" required>
  ID do badge
</ParamField>

<ParamField body="event_id" type="string" required>
  ID do evento
</ParamField>

<ParamField body="recipient_email" type="string" required>
  Email do recebedor
</ParamField>

<ParamField body="recipient_name" type="string" required>
  Nome do recebedor
</ParamField>

<ParamField body="fields" type="object">
  Campos personalizados do certificado
</ParamField>

## Response

<ResponseField name="post_id" type="string">
  ID do certificado criado
</ResponseField>

<ResponseField name="message" type="string">
  Mensagem de confirmação da emissão
</ResponseField>

<ResponseField name="recipient" type="object">
  Informações do recebedor
  <Expandable title="properties">
    <ResponseField name="email" type="string">
      Email do recebedor
    </ResponseField>

    <ResponseField name="name" type="string">
      Nome do recebedor
    </ResponseField>
  </Expandable>
</ResponseField>

<ResponseField name="badge" type="object">
  Informações do badge
  <Expandable title="properties">
    <ResponseField name="badge_id" type="string">
      ID do badge
    </ResponseField>

    <ResponseField name="name" type="string">
      Nome do badge
    </ResponseField>
  </Expandable>
</ResponseField>

<ResponseField name="event" type="object">
  Informações do evento
  <Expandable title="properties">
    <ResponseField name="event_id" type="string">
      ID do evento
    </ResponseField>

    <ResponseField name="name" type="string">
      Nome do evento
    </ResponseField>
  </Expandable>
</ResponseField>
