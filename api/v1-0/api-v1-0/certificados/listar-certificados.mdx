---
api: "GET /api/v1.0/assertions"
title: "Listar Certificados"
---

Lista certificados emitidos.

<Request method="GET" url="/api/v1.0/assertions" auth="bearer" />

## Path params

<ParamField path="userCode" type="string" required placeholder="ABC123XYZ">
  Código do usuário usado no path da rota.
</ParamField>

## Query params

<ParamField query="limit" type="number">
  Limite de resultados
</ParamField>

<ParamField query="offset" type="number">
  Offset para paginação
</ParamField>

## Response

<ResponseField name="data" type="array">
  Lista de certificados emitidos.
  <Expandable title="items">
    <ResponseField name="post_id" type="string">
      ID do certificado/post
    </ResponseField>

    <ResponseField name="created_at" type="string">
      Data/hora de emissão (ISO 8601)
    </ResponseField>

    <ResponseField name="badge" type="object">
      Informações do badge
      <Expandable title="properties">
        <ResponseField name="badge_id" type="string">
          ID do badge
        </ResponseField>

        <ResponseField name="name" type="string">
          Nome do badge
        </ResponseField>

        <ResponseField name="assets" type="object">
          Assets do badge
          <Expandable title="properties">
            <ResponseField name="image_url" type="string">
              URL da imagem do badge
            </ResponseField>
          </Expandable>
        </ResponseField>
      </Expandable>
    </ResponseField>

    <ResponseField name="profile" type="object">
      Perfil do destinatário
      <Expandable title="properties">
        <ResponseField name="name" type="string">
          Nome
        </ResponseField>

        <ResponseField name="nickname" type="string">
          Apelido/username
        </ResponseField>

        <ResponseField name="company_name" type="string">
          Nome da empresa
        </ResponseField>
      </Expandable>
    </ResponseField>

    <ResponseField name="event" type="object">
      Evento relacionado
      <Expandable title="properties">
        <ResponseField name="event_id" type="string">
          ID do evento
        </ResponseField>

        <ResponseField name="name" type="string">
          Nome do evento
        </ResponseField>

        <ResponseField name="description" type="string">
          Descrição do evento
        </ResponseField>

        <ResponseField name="category" type="string">
          Categoria do evento
        </ResponseField>
      </Expandable>
    </ResponseField>

    <ResponseField name="winner" type="object">
      Informações da premiação
      <Expandable title="properties">
        <ResponseField name="reason" type="string">
          Motivo/justificativa da conquista
        </ResponseField>
      </Expandable>
    </ResponseField>
  </Expandable>
</ResponseField>

<ResponseField name="total" type="number">
  Total de resultados encontrados
</ResponseField>

<ResponseField name="limit" type="number">
  Limite de resultados por página
</ResponseField>

<ResponseField name="offset" type="number">
  Offset usado na paginação
</ResponseField>
