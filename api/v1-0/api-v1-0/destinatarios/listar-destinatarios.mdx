---
api: "GET /api/v1.0/recipients"
title: "Listar Destinatários"
---

Lista destinatários.

<Request method="GET" url="/api/v1.0/recipients" auth="bearer" />

## Query params

<ParamField query="issuerId" type="string">
  ID do emissor (opcional)
</ParamField>

<ParamField query="limit" type="number">
  Limite de resultados
</ParamField>

<ParamField query="offset" type="number">
  Offset para paginação
</ParamField>

## Response

<ResponseField name="data" type="array">
  Lista de destinatários.
  <Expandable title="items">
    <ResponseField name="recipient_id" type="string">
      ID do destinatário
    </ResponseField>

    <ResponseField name="reason" type="string">
      Motivo/justificativa do recebimento
    </ResponseField>

    <ResponseField name="created_at" type="string">
      Data/hora de criação (ISO 8601)
    </ResponseField>

    <ResponseField name="event" type="object">
      Evento relacionado
      <Expandable title="properties">
        <ResponseField name="event_id" type="string">ID do evento</ResponseField>
        <ResponseField name="name" type="string">Nome do evento</ResponseField>
        <ResponseField name="description" type="string">Descrição do evento</ResponseField>
        <ResponseField name="category" type="string">Categoria do evento</ResponseField>
      </Expandable>
    </ResponseField>

    <ResponseField name="profile" type="object">
      Dados do perfil do destinatário
      <Expandable title="properties">
        <ResponseField name="name" type="string">Nome</ResponseField>
        <ResponseField name="nickname" type="string">Apelido/username</ResponseField>
        <ResponseField name="email" type="string">Email</ResponseField>
      </Expandable>
    </ResponseField>

    <ResponseField name="badge" type="object">
      Badge/conquista vinculada
      <Expandable title="properties">
        <ResponseField name="badge_id" type="string">ID do badge</ResponseField>
        <ResponseField name="name" type="string">Nome do badge</ResponseField>

        <ResponseField name="assets" type="object">
          Assets do badge
          <Expandable title="properties">
            <ResponseField name="image_url" type="string">
              URL da imagem do badge
            </ResponseField>
          </Expandable>
        </ResponseField>
      </Expandable>
    </ResponseField>
  </Expandable>
</ResponseField>

<ResponseField name="total" type="number">
  Total de resultados encontrados
</ResponseField>

<ResponseField name="limit" type="number">
  Limite de resultados por página
</ResponseField>

<ResponseField name="offset" type="number">
  Offset usado na paginação
</ResponseField>
